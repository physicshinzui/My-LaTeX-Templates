}

\begin{dfn}[Cauchy sequence]
For every $\epsilon > 0$, there exists $n_0 \in \mathbb{N}$ such that for every $m, n \ge n_0 \ (m,n \in \mathbb{N}),  \lvert a_m  - a_n  \rvert < \epsilon $.
\end{dfn}
